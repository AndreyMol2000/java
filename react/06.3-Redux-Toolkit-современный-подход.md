  name: "users",
  initialState: { data: [], loading: false, error: null },
  reducers: {},
  extraReducers: (builder) => {

        state.loading = false;
        state.data = action.payload;
      })

        state.loading = false;
        state.error = action.error.message;
      });
  }
});

export default usersSlice.reducer;
